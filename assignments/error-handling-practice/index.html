SOLUTIONS TO ERROR HANDLING PRACTICE CHALLENGES

Challenge 1: Sum Function with Type Validation

Part 1a Solution: Function Definition

function sum(x, y) {
  // Check data types first and throw error
  if (typeof x !== 'number' || typeof y !== 'number') {
    throw new TypeError('Both arguments must be numbers');
  }

  return x + y;
}

Part 1b Solution: Calling with Error Handling

try {
  // This will throw an error since we're passing strings
  const result = sum("1", "2");
  console.log("Result:", result); // This line won't execute
} catch (error) {
  console.log("Error caught:", error.message);
}

Challenge 2: User Authentication with Error Handling

Part 2a Solution: Login Function Definition

var user = {username: "sam", password: "123abc"};

function login(username, password) {
  // Check credentials
  if (username !== user.username || password !== user.password) {
    throw new Error("Invalid username or password");
  }

  console.log("Login successful!");
}

Part 2b Solution: Calling with Error Handling

// Example with correct credentials
try {
  login("sam", "123abc");
} catch (error) {
  console.log("Login error:", error.message);
}

// Example with incorrect credentials
try {
  login("sam", "wrongpassword");
} catch (error) {
  console.log("Login error:", error.message);
}
