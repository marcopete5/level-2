6. Applying CRUD with a Simple Server

CRUD operations come to life in a REST API server. Using Node.js and Express, you can handle all CRUD requests for a resource like tasks.

Example: Basic CRUD Server

This server supports CRUD operations for a task list stored in memory. 


const express = require('express');
const app = express();
app.use(express.json());
let tasks = [];

app.post('/tasks', (req, res) => {
    const task = { id: tasks.length + 1, ...req.body };
    tasks.push(task);
    res.status(201).json(task);
});

app.get('/tasks', (req, res) => res.json(tasks));

app.put('/tasks/:id', (req, res) => {
    const id = parseInt(req.params.id);
    const task = tasks.find(t => t.id === id);
    if (task) {
        Object.assign(task, req.body);
        res.json(task);
    } else {
        res.status(404).json({ error: 'Task not found' });
    }
});

app.delete('/tasks/:id', (req, res) => {
    tasks = tasks.filter(t => t.id !== parseInt(req.params.id));
    res.status(204).send();
});

app.listen(3000, () => console.log('Server running on port 3000'));
